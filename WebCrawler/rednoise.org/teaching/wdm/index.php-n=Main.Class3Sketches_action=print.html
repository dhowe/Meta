In-Class Sketches

// Load-Font, Split & Move

import rita.*;

RiText rt, rts[];

String s = "Green Ideas Sleep Furiously";

void setup() { 

  size(600, 400);
  RiText.setDefaultFont("AcademyEngravedLetPlain-48.vlw");  
  rt = new RiText(this, s, 20, 100);
  rts = rt.split();
  RiText.delete(rt); 

  rts[1].moveTo(20,150,1,3);
  rts[2].moveTo(20,250,1,3);
  rts[3].moveTo(20,200,,3);

}

void draw() { 

  background(255);

}

// Create-Font, Class & Object

import rita.*;

ShadowRiText srt;

void setup() { 

  size(400, 400);

  RiText.setDefaultColor(200,0,0);
  RiText.createDefaultFont("Arial",46);

  // create an instance of our class
  srt = new ShadowRiText(new RiText(this, "click"));

}

void mouseClicked() { 

  srt.moveTo(290, 390, 3);

}

void draw() { 

  background(255);

}

// define a custom class

class ShadowRiText { 

  RiText primary, shadow;
  ShadowRiText(RiText rt) {
    shadow = rt;
    primary = rt.copy();
    shadow.fill(40);
    shadow.x += 2; 
    shadow.y += 2; 
  }
  void moveTo(float x, float y, float secs) {
    primary.moveTo(x, y, secs);
    shadow.moveTo(x+2, y+2, secs);
  }

}// end class

// Create/Load & Switch Fonts

import rita.*;

RiText rt; PFont pf1, pf2; int clicks= 0;

void setup() { 

  size(400, 400);
  RiText.setDefaultAlignment(CENTER);
  pf1 = loadFont("SnellRoundhand-Bold-48.vlw");  
  pf2 = createFont("Arial",46);    
  rt = new RiText(this, "click");
  rt.textFont(pf2);

}

void mouseClicked() { 

  // switch btwn pf1 and pf2 each click
  PFont next = ++clicks%2==1 ? pf1 : pf2;
  rt.textFont(next);

}

void draw() { 

  background(255);

}
