  import java.text.DecimalFormat;
  import processing.core.*;
  import rita.*;

  /*
   * This is a template you can use (if you like) to
   * help with testing your Concordance code. If you use
   * it, be sure you are updated to RiTa v44 or higher. 
   *
   * In either case, your class should supply at least
   * the following methods:
   *     
   *
   *   void addTokens(String[] words);
   *
   *   int getFrequency(String word);
   *
   *   float getProbability(String word);  // btwn 0-1
   *
   *   String[] getMostCommonTokens(int count);
   *
   */
  public class MyConcorderApplet extends PApplet
  { 
    DecimalFormat DF = new DecimalFormat("#.###");

    RiText[] rts;  
    String[] mostCommon;
    float[] bestProbs;

    public void setup() 
    {
      size(550,550);

      // *** Replace w' your version ***
      MyConcorder rc =  new MyConcorder(this); 

      // and (optionally) support these options
      rc.setIgnoreCase(true);
      rc.setIgnorePunctuation(true);
      rc.setIgnoreStopWords(false);

      // Load each line from a test file
      String[] lines = loadStrings("examples/myTestFile.txt");

      // Split each line into individual words and add to model
      RiTokenizer rt = RiTokenizer.getWordTokenizer(this);    
      for (int i = 0; i < lines.length; i++) 
      {
        String[] words = rt.tokenize(lines[i]);

        // now add words to your concorder
        rc.addTokens(words);
      }

      // ask your model for the best 'n' tokens
      mostCommon = rc.getMostCommonTokens(15);

      // then get the probability for each
      bestProbs = new float[mostCommon.length]; 
      for (int i = 0; i < mostCommon.length; i++)
        bestProbs[i] = rc.getProbability(mostCommon[i]);    
    }

    // Display the result as you like
    public void draw() {
      background(255);
      if (rts == null)
        createRiTexts(); 
    }

    void createRiTexts()
    {
      RiText.setDefaultFont("Ziggurat32.vlw");
      rts = new RiText[mostCommon.length];
      int yOff = 30;
      for (int i = 0; i < mostCommon.length; i++) {
        String text = mostCommon[i] + " : " + DF.format(bestProbs[i]);
        rts[i] = new RiText(this, text); 
        rts[i].setY(yOff += 30);
      }
    }
  }
