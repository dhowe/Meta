Main: NGramModelIF

  import java.util.Map;

  /**
   * An interface to support different N-Gram model implementations 
   */
  public interface NGramModelIF
  {
    /** 
     * Sets the N-factor for the model 
     */
    void setNFactor(int n);

    /** 
     * Loads the <code>text</code> from which to build the model 
     */
    void loadModel(String text);

    /** 
     * Probability (0-1) of a single <code>letter</code> (1-gram) 
     */
    float getProbability(char letter);

    /** 
     * Probability of a sequence of k letters where k <= n (n-gram)
     */
    float getProbability(char[] sequence);

    /** 
     * Probabilities for all possible next letters after the
     * <code>sequence</code> (you may assume sequence.length < n)
     */
    Map getAllProbabilities(char[] sequence); 

    /** 
     * Generates, using probabilistic random selection, the 
     * next letter after the given <code>sequence</code> 
     * (you may assume sequence.length < n)
     */  
    String generateNext(char[] sequence); 

    /** 
     * Generates, according to the model's probabilities, a String
     * of <code>length</code> letters. 
     * @see #generateNext(char[]) 
     */
    String generateSequence(int length); 

  }// end
