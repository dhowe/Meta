// Function to fetch 'titles' from Google news

String[] getNewsTitles(String query) {

  JSONObject json = loadJSONObject
    ("https://ajax.googleapis.com/ajax/services/search/news?v=1.0&q="+query);
  JSONObject rData = json.getJSONObject("responseData");
  JSONArray results = rData.getJSONArray("results");
  String[] titles = new String[results.size()];
  for (int i = 0; i < results.size(); i++) {
    JSONObject item = results.getJSONObject(i);
    String title = item.getString("title");
    titles[i] = title;
  }
  return titles;
}

String query = "Writing Digital Media";

String url = api.replace("%QUERY%", query).replaceAll(" ","+");
JSONObject json = loadJSONObject(url);

int totalItems = json.getInt("totalItems");
JSONArray items = json.getJSONArray("items");
JSONObject item = items.getJSONObject(0);
JSONObject info = item.getJSONObject("volumeInfo");

println(totalItems + " items found");
println("#1: "+info.getString("title"));
println(info.getString("description"));

// Processing sketch using the 'substitute' function
// See: http://rednoise.org/wdm/substitutor/

String choice = "Click for a Variation...";
String source = "We|You|He|She chewed on the tire|bone as if it was chewing_gum|heaven|my_head.";


void setup() {

  size(500, 100);
  textAlign(CENTER);
  textFont(createFont("Times", 24));
}

void draw() {

  background(0);
  text(choice, width/2, height/2+12);
}

void mouseClicked() {

  choice = substitute(source);
}



/////////////////////////////////////////////////////////////////////

String substitute(String phrase) {

    if (phrase.indexOf('|') < 0)
      return phrase;

    phrase = phrase.trim()
        .replaceAll("[\\s]+"," ")
        .replaceAll("[\\t\\r\\n]+", "");

    String[] words = phrase.split(" ");
    for (int i = 0; i < words.length; i++) {

        if (words[i].indexOf('|') > -1) {

            boolean endsWithPunc = false;
            char last = words[i].charAt(words[i].length()-1);

            if ("?;:.!".indexOf(last) > -1) {

                endsWithPunc = true;
                words[i] = words[i].substring(0,words[i].length()-1);
            }

            String[] opts = words[i].split("\\|");
            words[i] = opts[(int)(Math.random()*opts.length)];

            if (words[i] != null && words[i].length()>0)
                if (endsWithPunc) words[i] += last;
        }
    }

    String result = "";
    for (int i = 0; i < words.length; i++)
      result += words[i] + " ";

    return result.trim()
        .replaceAll("\\s\\s+", " ")
        .replaceAll("_+", " ");
}
