
// Do a search on Google books 

String api = "https://www.googleapis.com/books/v1/volumes?q=%QUERY%";
String query = "Writing Digital Media";

String url = api.replace("%QUERY%", query).replaceAll(" ","+");
JSONObject json = loadJSONObject(url);

int totalItems = json.getInt("totalItems");
JSONArray items = json.getJSONArray("items");
JSONObject item = items.getJSONObject(0);
JSONObject info = item.getJSONObject("volumeInfo");

println(totalItems + " items found");
println("#1: "+info.getString("title"));
println(info.getString("description"));
