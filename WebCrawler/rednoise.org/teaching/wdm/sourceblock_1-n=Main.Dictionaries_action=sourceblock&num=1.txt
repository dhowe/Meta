
IntDict dict = new IntDict();
dict.set("c", 1);
dict.set("d", 2);

// Get the current value for "d"
int dcount = dict.get("d");

println("d: "+dcount);

// Add one to the value for "d"
dict.increment("d");

// Loop over all keys/values
for (String key : dict.keys()) 
{
  int count = dict.get(key);
  println(key+": "+count);
}
